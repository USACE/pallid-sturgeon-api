CREATE TABLE "CHANGE_TRACKING" 
   (	"TRK_ID" NUMBER, 
	"TRK_DATE" DATE, 
	"PSPA_TABLE" NUMBER, 
	"TBL_COLUMN" NUMBER, 
	"TBL_RECORD" VARCHAR2(20 BYTE), 
	"ORIG_VALUE" VARCHAR2(100 BYTE), 
	"NEW_VALUE" VARCHAR2(100 BYTE), 
	"TRK_COMMENTS" VARCHAR2(2000 BYTE), 
	"USERNAME" VARCHAR2(100 BYTE), 
	"SITE_YEAR" NUMBER, 
	 CONSTRAINT "CHANGE_TRACKING_PK" PRIMARY KEY ("TRK_ID")
   );

CREATE OR REPLACE TRIGGER "BI_CHANGE_TRACKING" 
  before insert on "CHANGE_TRACKING"               
  for each row
begin   
  if :NEW."TRK_ID" is null then 
    select "CHANGE_TRACKING_SEQ".nextval into :NEW."TRK_ID" from dual; 
  end if; 
end; 
/
ALTER TRIGGER "BI_CHANGE_TRACKING" ENABLE;
